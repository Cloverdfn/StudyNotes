### 题目描述

* 给定误差，用递推的梯形公式求出积分

### 代码实现

    #include<iostream>
    #include<fstream>
    #include<stdlib.h>
    #include<math.h>
    using namespace std;
    int main()
    {
        int n,k;
        double t,x,h,a,b,y,nu,s,y1;
        double u1,u;
        t=0;
        x=1;
        n=1;
        cin>>a>>b>>nu;
        h=(b-a)/nu;
        while(1)
        {
            x=a+n*h;
            t=x/(4+x*x)+t;
            if(7==n)
                break;
            n++;
        }
        y=((h/2)*(a/(4+a*a)+2*t+b/(4+b*b)));
        cout<<"T="<<y<<endl;
        x=1;
        s=0;
        k=0;
        u=1.0/2;
        while(1)
        {
            x=a+k*h+h*u;
            s=x/(4+x*x)+s;
            if(7==k)
                break;
            k++;

        }
        y1=((h/6)*(a/(4+a*a)+2*t+b/(4+b*b)+4*s));
        cout<<"S="<<y1<<endl;
        return 0;
    }
